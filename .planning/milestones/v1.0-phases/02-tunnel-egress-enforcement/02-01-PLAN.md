```go
package network

import "fmt"

type NetworkErrorType string

const (
    ErrBindingMissing    NetworkErrorType = "net.binding_missing"
    ErrEgressIPMismatch  NetworkErrorType = "net.egress_ip_mismatch"
    ErrDNSLeak           NetworkErrorType = "net.dns_leak"
    ErrLeakNotBlocked    NetworkErrorType = "net.leak_not_blocked"
    ErrEgressUnreachable NetworkErrorType = "net.egress_unreachable"
    ErrTunnelSetupFailed NetworkErrorType = "net.tunnel_setup_failed"
)

type NetworkError struct {
    Type     NetworkErrorType
    Message  string
    HostID   string
    Metadata map[string]any
}

func (e *NetworkError) Error() string {
    return fmt.Sprintf("[%s] %s", e.Type, e.Message)
}

func (e *NetworkError) EventType() string {
    return string(e.Type)
}

func (e *NetworkError) EventMetadata() map[string]any {
    if e.Metadata == nil {
        return map[string]any{"host_id": e.HostID}
    }
    m := make(map[string]any, len(e.Metadata)+1)
    for k, v := range e.Metadata {
        m[k] = v
    }
    m["host_id"] = e.HostID
    return m
}
```

**1. 创建 `internal/network/validate.go`：**

定义 `EgressValidator` 接口和 `ValidateEgressBinding` 函数：

```go
package network

import "context"

type EgressValidator interface {
    GetEgressIPByHost(ctx context.Context, hostID string) (EgressIPRecord, error)
    GetHostWgKeys(ctx context.Context, hostID string) (wgPrivateKey string, wgPublicKey string, err error)
}

type EgressIPRecord struct {
    ID             string
    IPAddress      string
    WgEndpoint     *string
    WgPublicKey    *string
    WgPresharedKey *string
    WgAllowedIPs   string
    WgDNSServer    *string
    WgPeerAddress  *string
}

func ValidateEgressBinding(ctx context.Context, v EgressValidator, hostID string) (EgressConfig, error) {
    // 1. 查询主机绑定的出口 IP
    record, err := v.GetEgressIPByHost(ctx, hostID)
    // err 或 record 不存在 → 返回 &NetworkError{Type: ErrBindingMissing, HostID: hostID}

    // 2. 校验隧道配置完整性
    // record.WgEndpoint == nil 或 record.WgPublicKey == nil
    //   → 返回 &NetworkError{Type: ErrTunnelSetupFailed, Message: "incomplete tunnel config"}

    // 3. 查询主机 WireGuard 密钥
    privKey, _, err := v.GetHostWgKeys(ctx, hostID)
    // privKey 为空 → 生成新密钥对（使用 wgtypes.GeneratePrivateKey()），
    //   调用 SetHostWgKeys 持久化（这需要额外接口方法，或先在 ValidateEgressBinding 内不处理生成逻辑，留给 PrepareHost 处理）

    // 4. 组装 EgressConfig 返回
    // 将 record 字段映射到 EgressConfig{ExpectedIP, Tunnel: TunnelSpec{...}}
}
```

注意：密钥生成涉及 `wgctrl` 依赖，此计划暂不引入该依赖（留给 02-02）。`ValidateEgressBinding` 只做绑定和配置完整性校验，密钥生成在 02-02 的 TunnelProvider 中处理。因此 `ValidateEgressBinding` 检查 `wg_private_key` 是否已存在于 hosts 表——如果为空，不报错但在返回的 `EgressConfig.Tunnel.PrivateKey` 中标记为空字符串，交由 PrepareHost 处理。

`EgressValidator` 接口使用独立的 `EgressIPRecord` 结构体（不直接依赖 repository.EgressIP），保持 network 包与 store 包的依赖方向清晰（network 不导入 store）。repository.Repository 实现此接口通过方法签名匹配。
