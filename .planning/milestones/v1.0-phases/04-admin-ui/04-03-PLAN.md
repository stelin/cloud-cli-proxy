**3. 创建 `admin_egress_ips.go`（新文件）：**

定义接口：
```go
type AdminEgressIPStore interface {
    ListEgressIPs(context.Context) ([]repository.EgressIP, error)
    GetEgressIP(context.Context, string) (repository.EgressIP, error)
    CreateEgressIP(context.Context, repository.CreateEgressIPParams) (repository.EgressIP, error)
    UpdateEgressIP(context.Context, string, repository.UpdateEgressIPParams) (repository.EgressIP, error)
    DeleteEgressIP(context.Context, string) error
}
```

`AdminEgressIPsHandler` struct 和方法：

a) `List()` — `GET /v1/admin/egress-ips`：返回 `{"egress_ips": [...]}`

b) `Get()` — `GET /v1/admin/egress-ips/{ipID}`：返回 `{"egress_ip": {...}}`

c) `Create()` — `POST /v1/admin/egress-ips`：
   - 解码 body：label（必填）、ip_address（必填，合法 IP）、provider（默认 "manual"）、wg_* 字段（可选）
   - wg_allowed_ips 默认 "0.0.0.0/0"
   - 返回 201 `{"egress_ip": {...}}`
   - IP 冲突返回 409

d) `Update()` — `PUT /v1/admin/egress-ips/{ipID}`：
   - 解码全量 body，更新所有字段
   - status 允许值：`available`, `disabled`
   - 返回 `{"egress_ip": {...}}`

e) `Delete()` — `DELETE /v1/admin/egress-ips/{ipID}`：
   - 调用 `store.DeleteEgressIP`
   - 如因 FK RESTRICT 失败，返回 409 `{"error": "egress IP is bound to a host, unbind first"}`
   - 成功返回 204

```sql

```

`AdminBindingsHandler` struct 和方法：

a) `Bind()` — `POST /v1/admin/bindings`：
   - 解码 body `{"host_id": "...", "egress_ip_id": "..."}`
   - **运行中主机保护检查**（per D-13）：调用 `store.GetHost(hostID)`，如 `host.Status == "running"` 返回 409 `{"error": "cannot bind egress IP to running host, stop host first"}`
   - 通过检查后调用 `store.BindEgressIPToHost`
   - 返回 201 `{"binding": {...}}`

b) `Unbind()` — `DELETE /v1/admin/bindings/{bindingID}`：
   - **运行中主机保护检查**（per D-14）：
     1. 调用 `store.GetBindingHostID(bindingID)` 获取关联 hostID
     2. 调用 `store.GetHost(hostID)`，如 `host.Status == "running"` 返回 409 `{"error": "cannot unbind egress IP from running host, stop host first"}`
   - 通过检查后调用 `store.UnbindEgressIPFromHost`
   - 返回 204

**1. 创建 `src/hooks/use-egress-ips.ts`** — TanStack Query hooks：

```typescript
interface EgressIP {
  id: string; label: string; ip_address: string; provider: string; status: string;
  wg_endpoint: string | null; wg_public_key: string | null; wg_preshared_key: string | null;
  wg_allowed_ips: string; wg_dns_server: string | null; wg_peer_address: string | null;
  created_at: string; updated_at: string;
}

export function useEgressIPs() {
  return useQuery({ queryKey: ["egress-ips"], queryFn: () => apiFetch<{ egress_ips: EgressIP[] }>("/egress-ips") });
}
export function useEgressIP(ipId: string) {
  return useQuery({ queryKey: ["egress-ips", ipId], queryFn: () => apiFetch<{ egress_ip: EgressIP }>(`/egress-ips/${ipId}`) });
}
export function useCreateEgressIP() {
  const qc = useQueryClient();
  return useMutation({
    mutationFn: (data: Partial<EgressIP>) => apiFetch<{ egress_ip: EgressIP }>("/egress-ips", { method: "POST", body: JSON.stringify(data) }),
    onSuccess: () => qc.invalidateQueries({ queryKey: ["egress-ips"] }),
  });
}
export function useUpdateEgressIP() {
  const qc = useQueryClient();
  return useMutation({
    mutationFn: ({ ipId, data }: { ipId: string; data: Partial<EgressIP> }) =>
      apiFetch<{ egress_ip: EgressIP }>(`/egress-ips/${ipId}`, { method: "PUT", body: JSON.stringify(data) }),
    onSuccess: () => qc.invalidateQueries({ queryKey: ["egress-ips"] }),
  });
}
export function useDeleteEgressIP() {
  const qc = useQueryClient();
  return useMutation({
    mutationFn: (ipId: string) => apiFetch(`/egress-ips/${ipId}`, { method: "DELETE" }),
    onSuccess: () => qc.invalidateQueries({ queryKey: ["egress-ips"] }),
  });
}
```

**2. 创建 `src/hooks/use-hosts.ts`**：

```typescript
interface HostWithUsername {
  id: string; user_id: string; status: string; template_image_ref: string;
  home_volume_name: string; slot_key: string; username: string;
  created_at: string; updated_at: string;
}
interface HostDetail {
  host: Host; user: User; bindings: { binding_id: string; egress_ip: EgressIP; created_at: string }[];
}

export function useHosts() {
  return useQuery({ queryKey: ["hosts"], queryFn: () => apiFetch<{ hosts: HostWithUsername[] }>("/hosts") });
}
export function useHostDetail(hostId: string) {
  return useQuery({ queryKey: ["hosts", hostId], queryFn: () => apiFetch<HostDetail>(`/hosts/${hostId}`) });
}
export function useHostAction() {
  const qc = useQueryClient();
  return useMutation({
    mutationFn: ({ hostId, action, body }: { hostId: string; action: "start" | "stop" | "rebuild"; body?: any }) =>
      apiFetch(`/hosts/${hostId}/${action}`, { method: "POST", ...(body ? { body: JSON.stringify(body) } : {}) }),
    onSuccess: () => {
      qc.invalidateQueries({ queryKey: ["hosts"] });
      qc.invalidateQueries({ queryKey: ["dashboard-stats"] });
    },
  });
}
export function useBindEgressIP() {
  const qc = useQueryClient();
  return useMutation({
    mutationFn: (data: { host_id: string; egress_ip_id: string }) =>
      apiFetch("/bindings", { method: "POST", body: JSON.stringify(data) }),
    onSuccess: () => qc.invalidateQueries({ queryKey: ["hosts"] }),
  });
}
export function useUnbindEgressIP() {
  const qc = useQueryClient();
  return useMutation({
    mutationFn: (bindingId: string) => apiFetch(`/bindings/${bindingId}`, { method: "DELETE" }),
    onSuccess: () => qc.invalidateQueries({ queryKey: ["hosts"] }),
  });
}
```
