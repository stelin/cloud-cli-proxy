```tsx
function StatusBadge({ status }: { status: string }) {
  switch (status) {
    case "passed":
      return <Badge className="bg-green-100 text-green-700 hover:bg-green-100">全部通过</Badge>;
    case "partial":
      return <Badge className="bg-yellow-100 text-yellow-700 hover:bg-yellow-100">部分通过</Badge>;
    case "failed":
      return <Badge className="bg-red-100 text-red-700 hover:bg-red-100">全部失败</Badge>;
    default:
      return <Badge variant="secondary">测试错误</Badge>;
  }
}
```

**4. StatusIcon 内部组件：**
```tsx
function StatusIcon({ status }: { status: string }) {
  switch (status) {
    case "pass":
      return <span className="flex h-5 w-5 items-center justify-center rounded-full bg-green-100"><Check className="h-3 w-3 text-green-600" /></span>;
    case "fail":
      return <span className="flex h-5 w-5 items-center justify-center rounded-full bg-red-100"><X className="h-3 w-3 text-red-600" /></span>;
    default:
      return <span className="flex h-5 w-5 items-center justify-center rounded-full bg-gray-100"><AlertCircle className="h-3 w-3 text-gray-500" /></span>;
  }
}
```

```typescript
function handleTest(ip: EgressIP) {
  setTestingId(ip.id);
  testMutation.mutate(ip.id, {
    onSuccess: (result) => {
      setTestResults((prev) => new Map(prev).set(ip.id, result));
      setTestDialogResult(result);
      setTestingId(null);
    },
    onError: () => {
      toast.error("测试失败");
      setTestingId(null);
    },
  });
}
```
