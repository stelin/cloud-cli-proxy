```go
func (r *Repository) ListHostsWithEgressByUserID(ctx context.Context, userID string) ([]UserHostSummary, error) {
    rows, err := r.db.Query(ctx, `
        SELECT h.id::text, h.hostname, h.status, COALESCE(host(e.ip_address), ''), h.created_at
        FROM hosts h
        LEFT JOIN host_egress_bindings b ON b.host_id = h.id
        LEFT JOIN egress_ips e ON e.id = b.egress_ip_id
        WHERE h.user_id = $1
        ORDER BY h.created_at ASC
    `, userID)
    // scan into []UserHostSummary, 每行一条
    // 注意：如果一台主机有多个绑定，SQL 会产生多行，但当前设计每台主机只有一个绑定
```

实现三个方法：

**List() — GET /v1/user/hosts (per D-04, D-05, D-12):**
- `userID := UserIDFromContext(r.Context())`
- 调用 `h.store.ListHostsWithEgressByUserID(ctx, userID)`
- 返回 `{"hosts": []UserHostSummary}`

**Get() — GET /v1/user/hosts/{hostID} (per D-04, D-11, D-13, D-14):**
- `userID := UserIDFromContext(r.Context())`
- `role := RoleFromContext(r.Context())`
- `hostID := r.PathValue("hostID")`
- 先调用 `h.store.GetHost(ctx, hostID)` 获取 host
- 归属校验：如果 `role != "admin" && host.UserID != userID`，返回 403 `{"error": "forbidden"}`
- 调用 `h.store.GetHostDetail(ctx, hostID)` 获取完整详情
- **过滤敏感字段**：构建 `UserHostDetail` 响应，bindings 只包含 `ip_address` 和 `tunnel_type`（不返回 wg_public_key、wg_preshared_key、proxy_config 等）
- 返回 `UserHostDetail` JSON

**Rebuild() — POST /v1/user/hosts/{hostID}/rebuild (per D-04, D-09, D-13, D-14):**
- `userID := UserIDFromContext(r.Context())`
- `role := RoleFromContext(r.Context())`
- `hostID := r.PathValue("hostID")`
- 先调用 `h.store.GetHost(ctx, hostID)` 做归属校验（同 Get）
- 调用 `h.queue.QueueHostAction(ctx, hostID, agentapi.ActionRebuildHost, userID)`
- 记录事件：type `user.host.rebuild`，message `用户发起主机重建`
- 返回 202 `{"task_id": task.ID, "status": "202 Accepted"}`
