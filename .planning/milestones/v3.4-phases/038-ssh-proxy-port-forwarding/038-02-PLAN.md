**Step 1 — 全局请求处理：**

```go
func (s *Server) handleGlobalRequests(reqs <-chan *ssh.Request, targetClient *ssh.Client) {
    for req := range reqs {
        switch req.Type {
        case "tcpip-forward", "cancel-tcpip-forward":
            ok, resp, err := targetClient.SendRequest(req.Type, req.WantReply, req.Payload)
            if err != nil {
                s.logger.Warn("forward global request to target failed", "type", req.Type, "error", err)
                if req.WantReply {
                    req.Reply(false, nil)
                }
                continue
            }
            if req.WantReply {
                req.Reply(ok, resp)
            }
        default:
            s.logger.Debug("unsupported global request", "type", req.Type)
            if req.WantReply {
                req.Reply(false, nil)
            }
        }
    }
}
```

**Step 2 — forwarded-tcpip 回传处理：**

```go
type forwardedTCPPayload struct {
    Addr       string
    Port       uint32
    OriginAddr string
    OriginPort uint32
}

func (s *Server) proxyForwardedChannels(incoming <-chan ssh.NewChannel, clientConn ssh.Conn, targetAddr string) {
    for newChan := range incoming {
        var payload forwardedTCPPayload
        if err := ssh.Unmarshal(newChan.ExtraData(), &payload); err != nil {
            s.logger.Warn("forwarded-tcpip payload unmarshal failed", "error", err)
            newChan.Reject(ssh.ConnectionFailed, "invalid payload")
            continue
        }

        s.logger.Debug("forwarded-tcpip channel received", "target", targetAddr, "addr", payload.Addr, "port", payload.Port)

        // 向客户端打开 forwarded-tcpip channel
        clientCh, clientReqs, err := clientConn.OpenChannel("forwarded-tcpip", newChan.ExtraData())
        if err != nil {
            s.logger.Warn("open forwarded-tcpip on client failed", "error", err)
            newChan.Reject(ssh.ConnectionFailed, "failed to open client channel")
            continue
        }
        go ssh.DiscardRequests(clientReqs)

        targetCh, targetReqs, err := newChan.Accept()
        if err != nil {
            s.logger.Error("accept forwarded-tcpip channel failed", "error", err)
            clientCh.Close()
            continue
        }
        go ssh.DiscardRequests(targetReqs)

        // 双向并发拷贝（标准模式）
        var wg sync.WaitGroup
        wg.Add(2)
        go func() {
            defer wg.Done()
            io.Copy(clientCh, targetCh)
            clientCh.CloseWrite()
        }()
        go func() {
            defer wg.Done()
            io.Copy(targetCh, clientCh)
            targetCh.CloseWrite()
        }()
        wg.Wait()

        s.logger.Debug("forwarded-tcpip channel closed", "target", targetAddr, "addr", payload.Addr, "port", payload.Port)
    }
}
```

修改 `internal/sshproxy/proxy.go` 的 `handleConnection`：

**Step 1 — 替换 `go ssh.DiscardRequests(globalReqs)`：**

将 `handleConnection` 中：
```go
go ssh.DiscardRequests(globalReqs)
```

替换为：在建立 `targetClient` 后启动 `handleGlobalRequests`。但 `targetClient` 的 dial 是 per-channel 的（当前架构），而 global requests 是 per-connection 的。因此需要调整：

**方案：** 在 `handleConnection` 中预先 dial 一次容器，建立共享的 `targetClient`，所有 channel 复用此连接。这是更合理的架构，且与 38-RESEARCH.md 的 Open Question 1 一致。

```go
func (s *Server) handleConnection(netConn net.Conn, config *ssh.ServerConfig) {
    defer netConn.Close()

    sshConn, chans, globalReqs, err := ssh.NewServerConn(netConn, config)
    if err != nil {
        s.logger.Debug("SSH handshake failed", "remote", netConn.RemoteAddr(), "error", err)
        return
    }
    defer sshConn.Close()

    ext := sshConn.Permissions.Extensions
    targetAddr := ext["target_addr"]
    targetUser := ext["target_user"]
    targetPassword := ext["target_password"]
    targetPrivateKey := ext["target_private_key"]
    s.logger.Info("SSH proxy session", "user", sshConn.User(), "target", targetAddr, "container_user", targetUser, "remote", netConn.RemoteAddr())

    // 预先建立到容器侧的共享 SSH 连接
    targetClient, err := s.dialContainer(targetAddr, targetUser, targetPassword, targetPrivateKey)
    if err != nil {
        s.logger.Error("dial container failed", "target", targetAddr, "error", err)
        return
    }
    defer targetClient.Close()

    // 启动全局请求处理 goroutine（SSH-02）
    go s.handleGlobalRequests(globalReqs, targetClient)

    // 注册 forwarded-tcpip handler（SSH-02）
    forwardedCh := targetClient.HandleChannelOpen("forwarded-tcpip")
    if forwardedCh != nil {
        go s.proxyForwardedChannels(forwardedCh, sshConn, targetAddr)
    } else {
        s.logger.Warn("forwarded-tcpip handler already registered, skipping")
    }

    // channel 分发循环
    for newChan := range chans {
        switch newChan.ChannelType() {
        case "session":
            go s.handleChannel(newChan, targetClient, targetAddr)
        case "direct-tcpip":
            go s.handleDirectTCPIP(newChan, targetClient, targetAddr)
        default:
            newChan.Reject(ssh.UnknownChannelType, fmt.Sprintf("channel type %s not supported", newChan.ChannelType()))
        }
    }
}
```
