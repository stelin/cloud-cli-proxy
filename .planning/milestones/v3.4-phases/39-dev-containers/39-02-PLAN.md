func runLocalDown(cmd *cobra.Command, args []string) error {
    cwd, _ := os.Getwd()
    mgr := local.NewLocalManager(local.LocalOptions{ProjectDir: cwd})
    if err := mgr.Down(cmd.Context()); err != nil {
        return err
    }
    fmt.Println("本地容器已停止并清理")

}

`localStatusCmd`（Use: "status", Short: "显示本地容器运行状态"）:
```go
func runLocalStatus(cmd *cobra.Command, args []string) error {
    cwd, _ := os.Getwd()
    mgr := local.NewLocalManager(local.LocalOptions{ProjectDir: cwd})
    status, err := mgr.Status(cmd.Context())
    if err != nil {
        return err
    }
    if status.Status == "not_found" {
        fmt.Println("本地容器未运行")
        return nil
    }
    fmt.Printf("容器: %s\n状态: %s\n镜像: %s\n端口: %s\n创建时间: %s\n",
        status.Name, status.Status, status.Image, status.PortMapping, status.CreatedAt)
    return nil
}
```

**internal/local/egress.go** — egress 配置处理:

```go
// EgressMode 表示 sing-box 启动模式
type EgressMode string
const (
    EgressModeTun   EgressMode = "tun"
    EgressModeProxy EgressMode = "proxy"
)

// DetectEgressMode 从 outbound JSON 检测协议类型
// socks/http → proxy 模式（macOS 兼容）
// 其他（vmess, vless, shadowsocks, trojan）→ tun 模式
func DetectEgressMode(outboundJSON []byte) (EgressMode, error)

// ValidateEgressConfig 验证 egress config 文件
// - 文件存在且可读
// - 是合法 JSON
// - 包含 "type" 字段
func ValidateEgressConfig(filePath string) ([]byte, EgressMode, error)

// egressMountArg 生成 docker create 的 bind mount 参数
// 将本地文件挂载到容器内 /etc/cloud-claude/sing-box-outbound.json
func egressMountArg(filePath string) string
```

**internal/local/local.go** — 在 Up 方法中集成 egress:

在 `buildCreateArgs` 中，如果 `opts.EgressConfig` 非空：
1. 调用 `ValidateEgressConfig(opts.EgressConfig)` 验证文件
2. 获取 egressMode
3. 在 docker create args 中追加：
   - `--cap-add NET_ADMIN`（tun 模式需要）
   - `--device /dev/net/tun`（tun 模式需要）
   - `-v {绝对路径}:/etc/cloud-claude/sing-box-outbound.json:ro`
   - `-e SING_BOX_MODE={egressMode}`

注意：proxy 模式不需要 `--cap-add NET_ADMIN` 和 `--device /dev/net/tun`，仅 tun 模式需要。
