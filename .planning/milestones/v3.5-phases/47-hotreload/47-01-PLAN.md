       ```go
       package tasks

       var (
         ErrBypassReloadInvalidInput = errors.New("bypass reload: missing snapshot id")
         ErrBypassReloadFailed       = errors.New("bypass reload: health check exhausted and no applied snapshot to rollback to")
       )

       // healthCheckRetries / healthCheckInterval / singboxReloadWait 是包级 var 便于测试调低。
       var (
         healthCheckRetries  = 3
         healthCheckInterval = 200 * time.Millisecond
         singboxReloadWait   = time.Second
       )

       // applyBypassRuleSetHook / verifyBypassHook / sleepHook 是测试注入点。
       var (
         applyBypassRuleSetHook = network.ApplyBypassRuleSet
         verifyBypassHook       = verifyBypassHealthyDefault // 私有 helper 默认实现：nsenter + curl 白名单 IP 测试
         sleepHook              = time.Sleep
       )

       func (w *Worker) handleReloadHostBypass(ctx context.Context, request agentapi.HostActionRequest) error {
         if request.BypassSnapshotID == "" { return ErrBypassReloadInvalidInput }
         snap, err := w.repo.GetBypassSnapshotByID(ctx, request.BypassSnapshotID)
         if err != nil { return fmt.Errorf("get snapshot: %w", err) }
         // 1. 下发新规则
         if err := applyBypassRuleSetHook(ctx, request.HostID, snap.WhitelistCIDRsJSON, snap.WhitelistDomainsJSON); err != nil {
           return w.markSnapshotFailedAndRollback(ctx, request, snap, err)
         }
         // 2. 等 sing-box watch reload
         sleepHook(singboxReloadWait)
         // 3. 健康检查 N 次
         for i := 0; i < healthCheckRetries; i++ {
           if checkErr := verifyBypassHook(ctx, request.HostID); checkErr == nil {
             _, _ = w.repo.UpdateBypassSnapshotStatus(ctx, snap.ID, "applied")
             w.recordReloadEvent(ctx, request, snap, "bypass.reload_applied", "")
             return nil
           } else if i < healthCheckRetries-1 {
             sleepHook(healthCheckInterval)
           }
         }
         // 4. 三次失败 → 自动 rollback
         return w.markSnapshotFailedAndRollback(ctx, request, snap, errors.New("health check exhausted"))
       }

       func (w *Worker) markSnapshotFailedAndRollback(ctx context.Context, request agentapi.HostActionRequest, current repository.BypassSnapshot, cause error) error {
         prev, prevErr := w.repo.GetLatestAppliedBypassSnapshot(ctx, request.HostID)
         if prevErr != nil {
           // 没有上一个 applied，无法 rollback —— 标记 failed 并 return
           _, _ = w.repo.UpdateBypassSnapshotStatus(ctx, current.ID, "failed")
           w.recordReloadEvent(ctx, request, current, "bypass.reload_failed", cause.Error())
           return ErrBypassReloadFailed
         }
         // 用 prev 重新下发，下发自身失败也只是 best-effort 告警
         if err := applyBypassRuleSetHook(ctx, request.HostID, prev.WhitelistCIDRsJSON, prev.WhitelistDomainsJSON); err != nil {
           w.recordReloadEvent(ctx, request, current, "bypass.reload_rollback_failed", err.Error())
         }
         _, _ = w.repo.UpdateBypassSnapshotStatus(ctx, current.ID, "rolled_back")
         w.recordReloadEvent(ctx, request, current, "bypass.reload_rolled_back", cause.Error())
         return nil // rollback 成功视为最终成功，避免 host 状态被翻为 failed
       }
       ```
