       ```go
       import (

       // ConfigureBypassFirewall 在已存在的 output 链上追加 v3.5 白名单 / uid 锁 / DNS 旁路阻断 /
       // 链末 log drop 一整套规则，并在 table 内创建空 set `whitelist_v4`（后续由 Phase 47 Plan 01

       // 调用顺序约束：必须在 applyWorkerIPv4Rules 已经添加完 lo / ct established / gwIP /
       // DNS 53 等基础规则之后调用。本函数追加的规则按 nft 自上而下匹配语义如下：
       //   先 drop（mDNS / LLMNR / NetBIOS），避免任何后续 accept 规则误放行 DNS 旁路；
       //   再 accept（tun / uid+443 / whitelist），分别覆盖 sing-box tun 出向 / 直连代理 / 白名单逃逸；
       //   最后链末 log drop，给 fail-closed 计数 + 排障证据。
       func ConfigureBypassFirewall(conn *nftables.Conn, table *nftables.Table, output *nftables.Chain, eth0IfIndex int, proxyIP net.IP, sshPort uint16) (*nftables.Set, error) {
         // 1. mDNS / LLMNR / NetBIOS UDP drop（5353 / 5355 / 137 / 138）
         for _, dport := range []uint16{5353, 5355, 137, 138} {
           addOifUDPDportDropRule(conn, table, output, eth0IfIndex, dport)
         }
         // 2. 白名单逃逸 tun
         addOifNameAcceptRule(conn, table, output, "sb-tun0")
         // 3. uid 锁 + proxyIP:443
         if proxyIP != nil {
           addOifSkuidIPPortAcceptRule(conn, table, output, eth0IfIndex, BypassSingboxUID, proxyIP, 443, ipprotoTCP)
         }
         // 4. 命名 set 匹配
         set := &nftables.Set{ Table: table, Name: BypassNftSetName, KeyType: nftables.TypeIPAddr, Interval: true }
         if err := conn.AddSet(set, nil); err != nil { return nil, fmt.Errorf("add nft set %s: %w", BypassNftSetName, err) }
         addOifNamedSetMatchAcceptRule(conn, table, output, eth0IfIndex, set)
         // 5. 链末 log drop
         addOifLogDropRule(conn, table, output, "sbfw-drop ")
         return set, nil
       }
       ```

## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| 容器进程 → nft output 链 | 进程层的 root + uid 切换都会被 meta skuid 匹配；nft 在内核态，容器内 root 无法 bypass |
| 容器进程 → DNS 旁路（mDNS/LLMNR/NetBIOS） | 即使白名单包含 224.0.0.251 等多播地址也优先 drop；防止 DNS 信号通过协议旁路逃逸 |
| nft chain 默认 policy → 链末 log drop | 默认 policy=drop 仅静默丢包；链末 log 是显式规则，可用 nft counter 计数 + 写 syslog 排障 |

|-----------|----------|-----------|-------------|-----------------|
