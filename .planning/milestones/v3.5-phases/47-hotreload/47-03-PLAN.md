       ```bash
           && pass "I10 SSH 不断（log lines>=5）" || skip "I10" "ssh loop 未启动"
       }

       # ─── 场景实现 ───
       run_scenario() {
         local name=$1
         echo "=== Scenario: $name ==="
         case "$name" in
           loopback-only) bind_preset loopback ;;
           lan-only) bind_preset lan ;;
           loopback-lan) bind_preset loopback; bind_preset lan ;;
           custom-ip) bind_rule '{"rule_type":"ip_cidr","value":"192.168.0.1/32"}' ;;
           custom-domain) bind_rule '{"rule_type":"domain_suffix","value":".internal.example.com"}' ;;
           fail-closed-pkill) bind_preset loopback ;;
         esac
         apply_and_wait
         # 通用断言
         assert_invariant_I1
         assert_invariant_I2
         assert_invariant_I3
         assert_invariant_I4
         assert_invariant_I6
         assert_invariant_I7
         assert_invariant_I8
         assert_invariant_I9
         assert_invariant_I10
         # 场景特定
         if [ "$name" = "fail-closed-pkill" ]; then
           assert_invariant_I5_fail_closed
           # BYPASS-VERIFY-04：fail-closed 后 snapshot.applied_status 必须能拉到 rolled_back
           assert_snapshot_state "rolled_back"
         else
           assert_snapshot_state "applied"
         fi
       }

       bind_preset() { api POST "/v1/admin/hosts/$TARGET_HOST_ID/bypass" "{\"preset_slug\":\"$1\"}"; }
       bind_rule() { api POST "/v1/admin/bypass/rules" "$1"; }
       apply_and_wait() {
         local res; res=$(api POST "/v1/admin/hosts/$TARGET_HOST_ID/bypass/apply" '{"note":"uat-bypass"}')
         local task_id; task_id=$(echo "$res" | jq -r '.task_id')
         # 轮询 task 完成（最多 30s）
         for i in $(seq 1 30); do
           local s; s=$(api GET "/v1/admin/tasks/$task_id" | jq -r '.status')
           case "$s" in
             succeeded|failed) return 0 ;;
           esac
           sleep 1
         done
       }
       assert_snapshot_state() {
         local want=$1
         local got; got=$(api GET "/v1/admin/hosts/$TARGET_HOST_ID/bypass/snapshots" | jq -r '.snapshots[0].applied_status')
         [ "$got" = "$want" ] && pass "BYPASS-VERIFY-04: snapshot.applied_status=$got" || fail "BYPASS-VERIFY-04: want=$want got=$got"
       }

       # ─── 主入口 ───
       if [ -z "$SCENARIO" ]; then
         echo "missing --scenario" >&2; usage; exit 2
       fi
       run_scenario "$SCENARIO"
       echo "── Summary: PASS=$PASS_COUNT FAIL=$FAIL_COUNT SKIP=$SKIP_COUNT ──"
       [ "$FAIL_COUNT" -eq 0 ] || exit 1
       exit 0
       ```
