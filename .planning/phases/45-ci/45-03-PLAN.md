<api_sketch>
<!-- waitfor.go 的目标 API 形态（具体字段命名可微调） -->

// WaitOpt 是 functional option，应用到内部 waitOptions 上。
type WaitOpt func(*waitOptions)

type waitOptions struct {
    timeout      time.Duration
    pollInterval time.Duration
    dumpHook     DumpHook
}

func WithTimeout(d time.Duration) WaitOpt
func WithPollInterval(d time.Duration) WaitOpt
func WithDumpHook(h DumpHook) WaitOpt   // Plan 04 注入；测试也用得上

// WaitFor 反复执行 predicate，直到：
//   - predicate 返回 nil → 立即返回 nil
//   - ctx.Done() → 立即返回 ctx.Err() 包装的错（不调 dump hook，因为这是上游主动取消，不是失败）
//   - timeout 到 → 调 dump hook → 返回 timed out 错（含 last predicate err 与 hook err）
//
// 默认 timeout=30s, pollInterval=500ms, dumpHook=NoopDumpHook。
func WaitFor(
    ctx context.Context,
    name string,
    predicate func(context.Context) error,
    opts ...WaitOpt,
) error

// ContainerHandle 抽象 testcontainers.Container 中本包需要的子集，
// 让 Plan 02 Scenario 句柄与 raw testcontainers.Container 都能传入。
type ContainerHandle interface {
    Logs(ctx context.Context) (io.ReadCloser, error)
    Exec(ctx context.Context, cmd []string) (int, io.Reader, error)
}

func WaitForLog(ctx context.Context, c ContainerHandle, substring string, opts ...WaitOpt) error
func WaitForPort(ctx context.Context, host string, port int, opts ...WaitOpt) error
func WaitForHTTP(ctx context.Context, url string, expectStatus int, opts ...WaitOpt) error
func WaitForExec(ctx context.Context, c ContainerHandle, cmd []string, expectExitCode int, opts ...WaitOpt) error
</api_sketch>

<polling_impl_note>
<!-- 实现轮询时必须用 ticker，不要 time.Sleep -->

正确范式：
    timer := time.NewTimer(opts.timeout)
    defer timer.Stop()
    ticker := time.NewTicker(opts.pollInterval)
    defer ticker.Stop()

    // 第一次立刻试一次（不等 ticker），后续按 ticker 周期
    var lastErr error
    for {
        lastErr = predicate(ctx)
        if lastErr == nil { return nil }
        select {
        case <-ctx.Done():
            return fmt.Errorf("waitfor name=%s: %w (last err: %v)", name, ctx.Err(), lastErr)
        case <-timer.C:
            // 超时：调 dump hook，组装错误
            hookErr := opts.dumpHook.OnWaitForTimeout(ctx, name, lastErr)
            // 组装 final error（见 <error_format>）
            return ...
        case <-ticker.C:
            // 继续下一轮
        }
    }

禁止：
    for {
        if err := predicate(ctx); err == nil { return nil }
        time.Sleep(opts.pollInterval)  // ❌ 会被 Plan 05 lint-no-bare-sleep 守护拦截
        ...
    }
</polling_impl_note>

       a) **包级默认值常量**：
          ```go
          const (
              DefaultWaitTimeout      = 30 * time.Second
              DefaultWaitPollInterval = 500 * time.Millisecond
          )
          ```

       b) **waitOptions + functional options**：
          ```go
          type waitOptions struct {
              timeout      time.Duration
              pollInterval time.Duration
              dumpHook     DumpHook
          }
          type WaitOpt func(*waitOptions)
          func WithTimeout(d time.Duration) WaitOpt       { return func(o *waitOptions) { o.timeout = d } }
          func WithPollInterval(d time.Duration) WaitOpt  { return func(o *waitOptions) { o.pollInterval = d } }
          func WithDumpHook(h DumpHook) WaitOpt           { return func(o *waitOptions) { o.dumpHook = h } }
          ```

          options 的默认初始化在 WaitFor 入口完成：
          ```go
          opts := waitOptions{timeout: DefaultWaitTimeout, pollInterval: DefaultWaitPollInterval, dumpHook: NoopDumpHook{}}
          for _, fn := range applyOpts { fn(&opts) }
          if opts.pollInterval <= 0 { opts.pollInterval = DefaultWaitPollInterval }  // 防御
          ```

       c) **核心 WaitFor**（严格按 `<polling_impl_note>` 模板）：

          - 入口先做一次 predicate（不等 ticker，避免 "已经就绪的对象还要等 500ms"）。
          - 用 `time.NewTimer(opts.timeout)` + `time.NewTicker(opts.pollInterval)`；**禁止** `time.Sleep`。
          - 三路 select：`<-ctx.Done()` → 立即返回 `ctx.Err()`（**不调 dump hook**，因为这是 caller 主动 cancel）；`<-timer.C` → 调 dump hook → 组装错误返回；`<-ticker.C` → 重新跑 predicate。
          - 超时分支错误组装严格遵守 `<error_format>`：
            ```go
            msg := fmt.Sprintf("waitfor name=%s: timed out after %s; last err: %v", name, opts.timeout, lastErr)
            if hookErr != nil {
                msg = fmt.Sprintf("%s; dump hook err: %v", msg, hookErr)
            }
            return errors.New(msg)
            ```

          ctx cancel 分支错误：
            ```go
            return fmt.Errorf("waitfor name=%s: %w (last err: %v)", name, ctx.Err(), lastErr)
            ```

          type ContainerHandle interface {
              Logs(ctx context.Context) (io.ReadCloser, error)
              Exec(ctx context.Context, cmd []string) (int, io.Reader, error)
          }

       e) **WaitForLog**（用 io.ReadAll + strings.Contains 的简单实现，O(N) 但够用）：
          ```go
          func WaitForLog(ctx context.Context, c ContainerHandle, substring string, opts ...WaitOpt) error {
              name := fmt.Sprintf("log_contains:%q", substring)
              return WaitFor(ctx, name, func(ctx context.Context) error {
                  r, err := c.Logs(ctx)
                  if err != nil { return fmt.Errorf("read logs: %w", err) }
                  defer r.Close()
                  b, err := io.ReadAll(r)
                  if err != nil { return fmt.Errorf("read logs body: %w", err) }
                  if !strings.Contains(string(b), substring) {
                      return fmt.Errorf("substring %q not found in logs (%d bytes)", substring, len(b))
                  }
                  return nil
              }, opts...)
          }
          ```

       f) **WaitForPort**（net.Dialer，1s 单次拨号超时，让 ticker 决定下一次重试）：
          ```go
          func WaitForPort(ctx context.Context, host string, port int, opts ...WaitOpt) error {
              addr := net.JoinHostPort(host, strconv.Itoa(port))
              name := fmt.Sprintf("port:%s", addr)
              dialer := &net.Dialer{Timeout: 1 * time.Second}
              return WaitFor(ctx, name, func(ctx context.Context) error {
                  conn, err := dialer.DialContext(ctx, "tcp", addr)
                  if err != nil { return err }
                  _ = conn.Close()
                  return nil
              }, opts...)
          }
          ```

       g) **WaitForHTTP**（标准 http.Client，2s 单次请求超时）：
          ```go
          func WaitForHTTP(ctx context.Context, url string, expectStatus int, opts ...WaitOpt) error {
              name := fmt.Sprintf("http:%s=%d", url, expectStatus)
              client := &http.Client{Timeout: 2 * time.Second}
              return WaitFor(ctx, name, func(ctx context.Context) error {
                  req, err := http.NewRequestWithContext(ctx, http.MethodGet, url, nil)
                  if err != nil { return fmt.Errorf("build req: %w", err) }
                  resp, err := client.Do(req)
                  if err != nil { return err }
                  defer resp.Body.Close()
                  if resp.StatusCode != expectStatus {
                      return fmt.Errorf("http %s: got %d want %d", url, resp.StatusCode, expectStatus)
                  }
                  return nil
              }, opts...)
          }
          ```
