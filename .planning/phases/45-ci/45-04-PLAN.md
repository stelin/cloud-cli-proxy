type ArtifactDumper struct {
    scenario *Scenario          // Plan 02 句柄；可为 nil（用例侧未注入 scenario 时不阻塞 dump）
    baseDir  string             // 形如 "./out/e2e-artifacts"；为空时走默认解析
    logger   *slog.Logger
}

// NewArtifactDumper 构造 dumper。
// baseDir 为空 → 用 EnvOrDefault("CLOUD_CLI_PROXY_E2E_ARTIFACT_DIR", "./out/e2e-artifacts")
// scenario 可为 nil，用于 unit test。
func NewArtifactDumper(scenario *Scenario, baseDir string) *ArtifactDumper

// Collect 在 baseDir/<sanitizedName>/<timestamp>/ 下建 5 个子目录占位，并在每个子目录
// 写入一份中文 README.md 说明 Phase 52 接入后的内容。
// 返回本次 dump 写入的根目录绝对路径（即 baseDir/<sanitizedName>/<timestamp>/）+ 错误。
//
// 关键属性：
//   - 幂等：同一 name + 同一 timestamp 多次调用，已存在目录与文件不报错（os.MkdirAll + 仅写入不存在的 README）
//   - 不污染：本 plan 的 unit test 用 t.TempDir() 作为 baseDir，结束后由 testing 框架自动清理
//   - 不真实执行：本 plan 不调 docker / nft / pg_dump
func (d *ArtifactDumper) Collect(ctx context.Context, name string) (string, error)

// OnWaitForTimeout 实现 DumpHook：
//   1. 调用 d.Collect(ctx, name)
//   2. 在 <dir>/system/wait-timeout.txt 追加一行：
//        <RFC3339 timestamp> name=<name> last_err=<lastErr.Error()>
//   3. 返回 Collect / WriteFile 任一错误（best-effort，hook 自身不 panic）
func (d *ArtifactDumper) OnWaitForTimeout(ctx context.Context, name string, lastErr error) error

// 包级 helper（导出方便 unit test）：
func sanitizeName(s string) string  // t.Name() 含 "/"，必须替换；空白替换为 "_"，避免 OS 路径问题
func defaultBaseDir() string         // 解析 env > 默认 "./out/e2e-artifacts"

       ```go
       //go:build e2e

       package harness

       import (
           "context"
           "fmt"
           "log/slog"
           "os"
           "path/filepath"
           "strings"
           "time"
       )

       // EnvArtifactBaseDir 环境变量可覆盖默认 artifact 根目录。
       const EnvArtifactBaseDir = "CLOUD_CLI_PROXY_E2E_ARTIFACT_DIR"

       // DefaultArtifactBaseDir 项目根相对路径，禁绝对路径（CONVENTIONS.md §Privacy）。
       const DefaultArtifactBaseDir = "./out/e2e-artifacts"

       // ArtifactSubdirs 与 REQUIREMENTS.md §OBS-02 五子目录契约保持一致。
       // Phase 52 OBS-01..03 接入后会在每个子目录写真实收集结果。
       var ArtifactSubdirs = []string{"logs", "network", "docker", "postgres", "system"}

       type ArtifactDumper struct {
           scenario *Scenario   // 可为 nil
           baseDir  string
           logger   *slog.Logger
       }

       func NewArtifactDumper(scenario *Scenario, baseDir string) *ArtifactDumper {
           if baseDir == "" {
               baseDir = defaultBaseDir()
           }
           return &ArtifactDumper{
               scenario: scenario,
               baseDir:  baseDir,
               logger:   slog.New(slog.NewTextHandler(os.Stderr, &slog.HandlerOptions{Level: slog.LevelInfo})),
           }
       }

       // Collect 在 baseDir/<sanitizedName>/<timestamp>/ 下建 5 个子目录与 README，
       // 返回该目录的绝对路径。
       func (d *ArtifactDumper) Collect(ctx context.Context, name string) (string, error) {
           timestamp := time.Now().UTC().Format("20060102T150405Z") // RFC3339-like，文件系统安全
           dir := filepath.Join(d.baseDir, sanitizeName(name), timestamp)
           if err := os.MkdirAll(dir, 0o755); err != nil {
               return "", fmt.Errorf("artifact: mkdir root: %w", err)
           }
           for _, sub := range ArtifactSubdirs {
               subDir := filepath.Join(dir, sub)
               if err := os.MkdirAll(subDir, 0o755); err != nil {
                   return dir, fmt.Errorf("artifact: mkdir %s: %w", sub, err)
               }
               readmePath := filepath.Join(subDir, "README.md")
               if _, err := os.Stat(readmePath); os.IsNotExist(err) {
                   if err := os.WriteFile(readmePath, []byte(readmeContentFor(sub)), 0o644); err != nil {
                       return dir, fmt.Errorf("artifact: write README %s: %w", sub, err)
                   }
               }
           }
           absDir, err := filepath.Abs(dir)
           if err != nil { return dir, fmt.Errorf("artifact: abs path: %w", err) }
           d.logger.Info("artifact collected", "name", name, "dir", absDir)
           return absDir, nil
       }

       // OnWaitForTimeout 实现 DumpHook（Plan 03 接口）。
       func (d *ArtifactDumper) OnWaitForTimeout(ctx context.Context, name string, lastErr error) error {
           dir, err := d.Collect(ctx, name)
           if err != nil { return fmt.Errorf("waitfor dump: %w", err) }
           note := fmt.Sprintf("%s name=%s last_err=%v\n",
               time.Now().UTC().Format(time.RFC3339), name, lastErr)
           notePath := filepath.Join(dir, "system", "wait-timeout.txt")
           f, err := os.OpenFile(notePath, os.O_APPEND|os.O_CREATE|os.O_WRONLY, 0o644)
           if err != nil { return fmt.Errorf("waitfor dump: open note: %w", err) }
           defer f.Close()
           if _, err := f.WriteString(note); err != nil {
               return fmt.Errorf("waitfor dump: append note: %w", err)
           }
           return nil
       }

       // sanitizeName 把 t.Name()（含 "/" 等）转成文件系统安全的目录名。
       func sanitizeName(s string) string {
           if s == "" { return "_unnamed" }
           r := strings.NewReplacer("/", "_", " ", "_", "\\", "_", ":", "_")
           return r.Replace(s)
       }

       // defaultBaseDir 解析环境变量 > 编译默认。
       func defaultBaseDir() string {
           if v := os.Getenv(EnvArtifactBaseDir); v != "" {
               return v
           }
           return DefaultArtifactBaseDir
       }

       // readmeContentFor 返回 <readme_template> 中对应子目录的占位文本。
       func readmeContentFor(sub string) string {
           switch sub {
           case "logs":
               return "# 容器日志（Phase 52 OBS-01 接入后真实收集）\n\nPhase 52 OBS-01 落地后，本目录会自动收集：\n- control-plane.log\n- host-agent.log\n- gateway-<host_id>.log\n- worker-<host_id>.log\n\n当前 Phase 45 Plan 04 仅创建本目录与 README 占位。\n"
           case "network":
               return "# 网络状态（Phase 52 OBS-02 接入后真实收集）\n\nPhase 52 OBS-02 落地后，本目录会包含：\n- nft-list-ruleset.txt\n- ip-netns.txt\n- ip-route-host.txt\n- ip-route-<host_id>.txt\n"
           case "docker":
               return "# Docker 元信息（Phase 52 OBS-02 接入后真实收集）\n\n- docker-ps.txt\n- docker-inspect-<name>.json\n"
           case "postgres":
               return "# Postgres dump（Phase 52 OBS-03 接入后真实收集）\n\n- pg-dump-hosts.csv 等关键表 CSV 导出\n- schema-version.txt\n"
           case "system":
               return "# 宿主机系统状态（Phase 52 OBS-02 接入后真实收集）\n\n- dmesg-tail.txt / proc-meminfo.txt / kernel-version.txt\n- wait-timeout.txt（Phase 45 Plan 04 已开始在写）\n"
           default:
               return "# Phase 45 Plan 04 占位\n"
           }
       }
       ```
