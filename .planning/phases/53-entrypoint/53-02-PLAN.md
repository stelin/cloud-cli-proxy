# Plan 53-02: entrypoint 重写

## Objective

把 `deploy/docker/managed-user/entrypoint.sh` 改造为 v4.0 fail-closed 启动序列：sing-box 通过 runuser 降权运行 → tun0 waitFor → nft default-deny → DNS 强制 sing-box stub → config 文件 rm → sing-box 死亡 PID 1 fail-closed → sshd foreground。

## Scope

| REQ | 实现位置 | 验收 |
|---|---|---|
| EP-01 | entrypoint 串行启动序列 | 任一步失败 entrypoint 非 0 退出 → 容器死 |
| EP-02 | `runuser -u singbox` 启动 sing-box | `ps -o uid,user,comm -p $(pidof sing-box)` = `uid=9000` |

## Tasks

### T1 — 拆出 `default-deny.nft` 规则集文件

新建 `deploy/docker/managed-user/default-deny.nft`：

```nft
#!/usr/sbin/nft -f
# v4.0 同容器架构 default-deny ruleset
# 等价于 v3.6 worker_firewall_linux.go::ApplyWorkerFirewallRules 输出链
# 但应用位置从 host netns 切到容器内自身 netns。
#
# 策略：默认 DROP，仅放行 tun0 出口 + lo + 反向已建立连接。
# DNS 强制走 sing-box stub resolver（127.0.0.1:53）—— D-V4-3。

table inet cloud_proxy_v4
delete table inet cloud_proxy_v4

table inet cloud_proxy_v4 {
    chain output {
        type filter hook output priority filter; policy drop;

        # 放行 lo（含 sing-box stub DNS 127.0.0.1:53）
        oifname "lo" counter accept

        # 放行 tun0（sing-box 接管）
        oifname "tun0" counter accept

        # 显式 drop 外部 DNS（防止用户 dig @8.8.8.8 旁路）
        # D-V4-3：外部 UDP/53、TCP/53、TCP/853 全 drop
        # 注：这条规则其实被 policy drop 兜底，显式写出来是为了 counter 可观测
        meta l4proto udp udp dport 53 ip daddr != 127.0.0.0/8 counter drop comment "extdns-udp-drop"
        meta l4proto tcp tcp dport { 53, 853 } ip daddr != 127.0.0.0/8 counter drop comment "extdns-tcp-drop"

        # 显式 drop link-local（v3.6 LEAK-07 同款）
        ip daddr 169.254.0.0/16 counter drop comment "linklocal-drop"

        # 反向连接放行
        ct state established,related counter accept

        # 兜底（policy drop 已覆盖，这条 counter 用于可观测）
        counter drop comment "default-deny-tail"
    }

    chain input {
        type filter hook input priority filter; policy drop;

        # 放行 lo
        iifname "lo" counter accept

        # 放行 SSH 入向（端口由 host-agent 配置，默认 22）
        tcp dport 22 counter accept comment "ssh-in"

        # 放行 KasmVNC 入向（MODE=remote 时）
        tcp dport 6080 counter accept comment "kasmvnc-in"

        # 反向连接放行
        ct state established,related counter accept

        counter drop comment "default-deny-input-tail"
    }
}
```

**Rationale**: 规则集独立成文件便于 review 和单测（Plan 53-03）；与 v3.6 worker_firewall ruleset 语义对齐（counter + link-local + default-deny）。

### T3 — entrypoint.sh 重大改造

**核心改造点**（保留 v3.x desktop / KasmVNC / 持久化逻辑，只在 sing-box 启动路径上动刀）：

```bash
#!/usr/bin/env bash
set -euo pipefail

# ... 保留 v3.x 头部的 LOG_DIR / write_desktop_config / prepare_v3_dirs 等函数 ...

# ===== v4.0: sing-box 同容器化启动序列 (PID 1 fail-closed) =====
# D-V4-1..4 + D-53-2/3/4/5/6 实现集中在这一段

SING_BOX_CONFIG="/etc/sing-box/config.json"
SING_BOX_USER="singbox"
NFT_RULESET="/etc/cloud-claude/default-deny.nft"
TUN_READY_TIMEOUT_S=30

start_singbox_or_die() {
  if [ ! -f "$SING_BOX_CONFIG" ]; then
    echo "[entrypoint] FATAL: sing-box config 不存在: $SING_BOX_CONFIG" >&2
    exit 1
  fi

  # 验证 config 文件权限（D-V4-2 / D-53-3 前置）
  perm="$(stat -c '%a' "$SING_BOX_CONFIG")"
  owner="$(stat -c '%U' "$SING_BOX_CONFIG")"
  if [ "$perm" != "600" ] || [ "$owner" != "root" ]; then
    echo "[entrypoint] FATAL: config 权限不对（want root:0600，got ${owner}:${perm}）" >&2
    exit 1
  fi

  # 启动 sing-box（runuser → singbox uid=9000，binary 文件 cap 提供 NET_ADMIN）
  echo "[entrypoint] starting sing-box as uid=9000 (file-cap based)"
  runuser -u "$SING_BOX_USER" -- /usr/local/bin/sing-box run -c "$SING_BOX_CONFIG" &
  SING_BOX_PID=$!

  # WaitFor tun0 ready（替代裸 sleep）
  echo "[entrypoint] waiting for tun0 (timeout=${TUN_READY_TIMEOUT_S}s)"
  local deadline=$((SECONDS + TUN_READY_TIMEOUT_S))
  while (( SECONDS < deadline )); do
    if ip link show tun0 >/dev/null 2>&1; then
      # 确认 sing-box 进程还在
      if kill -0 "$SING_BOX_PID" 2>/dev/null; then
        echo "[entrypoint] tun0 ready (sing-box pid=$SING_BOX_PID, uid=$(ps -o uid= -p "$SING_BOX_PID" | tr -d ' '))"
        return 0
      fi
    fi
    sleep 0.5
  done

  echo "[entrypoint] FATAL: tun0 未在 ${TUN_READY_TIMEOUT_S}s 内就绪" >&2
  kill "$SING_BOX_PID" 2>/dev/null || true
  exit 1
}

apply_nft_or_die() {
  echo "[entrypoint] applying nft default-deny ruleset"
  if ! nft -f "$NFT_RULESET"; then
    echo "[entrypoint] FATAL: nft 应用失败" >&2
    kill "$SING_BOX_PID" 2>/dev/null || true
    exit 1
  fi
  # 二次 verify
  if ! nft list table inet cloud_proxy_v4 >/dev/null 2>&1; then
    echo "[entrypoint] FATAL: nft table cloud_proxy_v4 未生效" >&2
    kill "$SING_BOX_PID" 2>/dev/null || true
    exit 1
  fi
}

lock_resolv_conf() {
  echo "[entrypoint] locking /etc/resolv.conf to sing-box stub"
  cat > /etc/resolv.conf <<'EOF'
# v4.0: DNS 强制走 sing-box stub resolver (D-V4-3)
nameserver 127.0.0.1
options edns0 trust-ad
EOF
  chmod 0644 /etc/resolv.conf
  # chattr +i 让用户无法修改（即便 user 是 root，没 cap_linux_immutable 也改不了）
  chattr +i /etc/resolv.conf 2>/dev/null || echo "[entrypoint] WARN: chattr +i 失败（可能是 overlayfs），依赖 nft 兜底"
}

remove_singbox_config() {
  # D-V4-2: sing-box 已加载到内存，从 fs 删除 config
  echo "[entrypoint] removing sing-box config from fs (D-V4-2)"
  shred -u "$SING_BOX_CONFIG" 2>/dev/null || rm -f "$SING_BOX_CONFIG"
  if [ -f "$SING_BOX_CONFIG" ]; then
    echo "[entrypoint] FATAL: config rm 失败" >&2
    kill "$SING_BOX_PID" 2>/dev/null || true
    exit 1
  fi
}

monitor_singbox_fail_closed() {
  # D-V4-4 / EP-04 / NET-03: sing-box 死 → 整个容器死
  (
    wait "$SING_BOX_PID" 2>/dev/null
    rc=$?
    echo "[entrypoint] FATAL: sing-box 退出 (exit=$rc)，触发容器死" >&2
    # 给 tini (PID 1) 发 TERM，让它清理子进程并退出
    kill -TERM 1 2>/dev/null
    # 兜底：如果 kill 1 没生效，硬退出
    sleep 2
    kill -KILL 1 2>/dev/null
  ) &
  echo "[entrypoint] singbox fail-closed monitor armed (pid=$!)"
}

# ===== 启动序列（fail-fast，任一步失败 entrypoint 退出 → tini 关停容器）=====

# v3.x 保留部分（仅 MODE != local 时跑 desktop 栈，与 v4.0 sing-box 启动并行）
# ... 此处保留 v3.x entrypoint L60-313 的 prepare_v3_dirs / KasmVNC / desktop 代码 ...

# v4.0 新增：sing-box 启动序列（所有 MODE 都跑）
start_singbox_or_die
lock_resolv_conf
apply_nft_or_die
remove_singbox_config
monitor_singbox_fail_closed

# v3.x SSH setup 保留（CONTAINER_USER 处理、密码、authorized_keys 等）
# 注意：v4.0 移除 sudoers 重写（v3.x L180-182 整段删除）
# ... 保留 v3.x L155-208 段 ...

# 兜底：删除 sudoers 残留（D-53-4）
rm -f /etc/sudoers.d/workspace /etc/sudoers.d/"${CONTAINER_USER:-workspace}" 2>/dev/null || true

# 删除 v3.x 旧的 MODE=local sing-box 分支（L316-384）—— 已被 v4.0 统一替换

# Foreground: sshd
exec /usr/sbin/sshd -D -e
```

### T5 — `chattr +i` 在 overlayfs 兜底策略

`chattr +i` 在某些 docker storage driver（overlay2）上可能返回 `Operation not supported`。

**对策**：
- entrypoint `chattr +i` 失败时仅 WARN 不 FATAL
- 主要依赖 nft 兜底：即便用户改 `/etc/resolv.conf` 指向 `8.8.8.8`，外部 DNS 流量也被 nft drop（T1 default-deny.nft `extdns-udp-drop` / `extdns-tcp-drop`）

**spike 验证项**：Plan 53-03 在 Ubuntu 24.04 + docker 24.x（默认 overlay2）上验证 chattr 行为，结果回填到 CONTEXT.md Open Question #4。
